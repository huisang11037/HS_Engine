144//
145// 함수: WndProc(HWND, UINT, WPARAM, LPARAM)
146//
147// 용도: 주 창의 메시지를 처리합니다.
148//
149// WM_COMMAND - 애플리케이션 메뉴를 처리합니다.
150// WM_PAINT - 주 창을 그립니다.
151// WM_DESTROY - 종료 메시지를 게시하고 반환합니다.
152//
153//
154LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
155{
156 switch (message)
157 {
158 case WM_COMMAND:
159 {
160 int wmId = LOWORD(wParam);
161 // 메뉴 선택을 구문 분석합니다:
162 switch (wmId)
163 {
164 case IDM_ABOUT:
165 DialogBox(hInst, MAKEINTRESOURCE(IDD_ABOUTBOX), hWnd, About);
166 break;
167 case IDM_EXIT:
168 DestroyWindow(hWnd);
169 break;
170 default:
171 return DefWindowProc(hWnd, message, wParam, lParam);
172 }
173 }
174 break;
175 case WM_PAINT:
176 {
177 PAINTSTRUCT ps;
178 // DC는 Device Context의 약자로, 그리기 작업을 수행하는데 필요한 정보를 담고 있는 구조체다.
179 HDC hdc = BeginPaint(hWnd, &ps);
180
181 // TODO: 여기에 hdc를 사용하는 그리기 코드를 추가합니다...
182 EndPaint(hWnd, &ps);
183 }
184 break;
185 case WM_DESTROY:
186 PostQuitMessage(0);
187 break;
188 default:
189 return DefWindowProc(hWnd, message, wParam, lParam);
190 }
191 return 0;
192}
